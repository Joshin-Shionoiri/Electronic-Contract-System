const customers = JSON.parse(localStorage.getItem("customers") || "[]"); function loadCustomers() { const select = document.getElementById("customerSelect"); select.innerHTML = ""; customers.forEach(c => { const opt = document.createElement("option"); opt.value = c.customerCode; // ← 新しいコード形式をそのまま使う opt.textContent = `${c.customerCode} ${c.customerName}`; select.appendChild(opt); }); }
